Create `src/app/[locale]/layout.tsx`. The file must be an async Server Component (no `'use client'` directive). It receives `params` as a Promise in Next.js 15:

```typescript
import type {Metadata} from 'next';
import {NextIntlClientProvider, hasLocale} from 'next-intl';
import {notFound} from 'next/navigation';
import {routing} from '@/i18n/routing';
import {spaceGrotesk, ibmPlexArabic} from '@/lib/fonts';
import '@/app/globals.css';

type Props = {
  children: React.ReactNode;
  params: Promise<{locale: string}>;
};

export function generateStaticParams() {
  return routing.locales.map((locale) => ({locale}));
}

export const metadata: Metadata = {
  title: 'ORKI — Underground Streetwear',
  description: 'Saudi streetwear. Dark. Underground.',
};

export default async function LocaleLayout({children, params}: Props) {
  // Await params — Next.js 15 params is a Promise
  const {locale} = await params;

  // Reject invalid locale values — hasLocale guard prevents open-redirect via locale segment
  if (!hasLocale(routing.locales, locale)) {
    notFound();
  }

  // Set direction from locale — BOTH lang and dir update atomically (non-negotiable architectural rule)
  const dir = locale === 'ar' ? 'rtl' : 'ltr';

  return (
    <html
      lang={locale}
      dir={dir}
      className={`dark ${spaceGrotesk.variable} ${ibmPlexArabic.variable}`}
    >
      <body className="bg-black text-white antialiased">
        <NextIntlClientProvider>
          {children}
        </NextIntlClientProvider>
      </body>
    </html>
  );
}
```

Key points:
- `className="dark"` is permanent — this site has no light mode (dark-only per CLAUDE.md)
- `spaceGrotesk.variable` and `ibmPlexArabic.variable` inject CSS variables onto `<html>` — globals.css :lang() selectors then apply the correct font by locale
- DO NOT use `spaceGrotesk.className` — that would bypass the :lang() switching strategy
- `lang` and `dir` are set in the same JSX expression — they always update together

If the `src/app/[locale]/` directory does not exist yet (create-next-app may have created `src/app/layout.tsx` instead), create the directory and move or replace the file. The create-next-app root `src/app/layout.tsx` should be deleted if a `src/app/[locale]/layout.tsx` is created — they cannot coexist for the same route.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| URL locale param → html[lang][dir] | Locale segment from URL controls HTML attributes rendered to browser |
| next/font → browser | Font files fetched by browser — controlled by Next.js preload injection |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | locale param → lang/dir attributes | mitigate | `hasLocale()` guard in layout.tsx: only `'en'` and `'ar'` pass through; any other value triggers `notFound()`; locale is never interpolated into raw CSS strings |
| T-03-02 | Spoofing | Font file delivery | accept | next/font self-hosts font files from Google Fonts at build time; no runtime CDN dependency; font files served from same origin |
| T-03-03 | Tampering | CSS variable injection via className | accept | Font CSS variables injected via `spaceGrotesk.variable` which is a stable hash-prefixed class generated by next/font — not user-controlled |
</threat_model>
