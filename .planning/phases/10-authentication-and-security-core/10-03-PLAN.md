```ts
/**
 * Auth Server Action result envelope.
 * Mirrors src/app/actions/cart.ts:30-37 with auth-specific codes.
 *
 * SEC-06 (no enumeration): wrong-email and wrong-password BOTH map to
 * INVALID_CREDENTIALS / 'Auth.errors.invalidCredentials'. Never differentiate.
 */
export type AuthActionResult<T = unknown> =
  | { ok: true; data: T }
  | {
      ok: false;
      code: 'INVALID_CREDENTIALS' | 'EMAIL_IN_USE' | 'WEAK_PASSWORD' | 'RATE_LIMITED' | 'VALIDATION' | 'UNKNOWN';
      fields?: Record<string, string>;
      messageKey: string;
    };

interface SupabaseAuthErrorShape {
  message?: string;
  status?: number;
  code?: string;
}

/**
 * Map any Supabase auth error to AuthActionResult code + messageKey.
 * Anti-enumeration: wrong-email and wrong-password BOTH produce INVALID_CREDENTIALS.
 * Existing-email signup is also collapsed to UNKNOWN (anti-enumeration); use the
 * audit log to distinguish when needed.
 */
export function mapSupabaseError(err: unknown): {
  code: Exclude<AuthActionResult['ok'] extends false ? AuthActionResult['code'] : never, never>;
  messageKey: string;
} {
  const e = (err ?? {}) as SupabaseAuthErrorShape;
  const msg = (e.message ?? '').toLowerCase();
  const status = e.status ?? 0;

  // Rate limit (429 from Supabase Auth or message contains 'rate')
  if (status === 429 || /rate.?limit/.test(msg) || /too.?many/.test(msg)) {
    return { code: 'RATE_LIMITED', messageKey: 'Auth.errors.tooManyAttempts' };
  }
  // Weak password (422)
  if (status === 422 || /password.*(short|weak|requirement)/.test(msg)) {
    return { code: 'WEAK_PASSWORD', messageKey: 'Auth.errors.passwordTooShort' };
  }
  // Invalid credentials — collapse wrong-email + wrong-password
  if (/invalid.*(login|credentials|email|password)/.test(msg)) {
    return { code: 'INVALID_CREDENTIALS', messageKey: 'Auth.errors.invalidCredentials' };
  }
  // Existing-email signup — UI-SPEC §Anti-patterns #11: do NOT surface 'EMAIL_IN_USE'
  // to the wire. Map to UNKNOWN; the auth_events row records the collision for ops.
  if (/already.*registered|user.*exists|email.*exists/.test(msg)) {
    return { code: 'UNKNOWN', messageKey: 'Auth.errors.unknown' };
  }
  return { code: 'UNKNOWN', messageKey: 'Auth.errors.unknown' };
}
```

Note: `EMAIL_IN_USE` exists in the union for completeness (e.g., it's allowed in the type system) but the mapper never returns it — anti-enumeration. The code value is reserved for any future internal-only branching.
</approach>

```ts
import 'server-only';
import { db } from '@/lib/db/client';
import { authEvents } from '@/lib/db/schema';

export type AuthEventType =
  | 'signup'
  | 'signin'
  | 'signin_failed'
  | 'password_reset_requested'
  | 'password_changed'
  | 'signout'
  | 'admin_action';

export async function writeAuthEvent(params: {
  type: AuthEventType;
  userId?: string | null;
  email?: string | null;
  metadata?: Record<string, unknown>;
  request?: Request;
}): Promise<void> {
  try {
    let ipAddress: string | null = null;
    let userAgent: string | null = null;
    if (params.request) {
      ipAddress = params.request.headers.get('x-forwarded-for')?.split(',')[0]?.trim()
              ?? params.request.headers.get('x-real-ip')
              ?? null;
      userAgent = params.request.headers.get('user-agent') ?? null;
    }
    await db.insert(authEvents).values({
      userId: params.userId ?? null,
      email: params.email ?? null,
      event: params.type,
      metadata: params.metadata ? (params.metadata as never) : null,
      ipAddress,
      userAgent,
    });
  } catch (err) {
    // Audit failures must NEVER break the user's auth flow.
    console.error('[writeAuthEvent]', err);
  }
}
```

1. **`signUpAction(input: SignupInput): Promise<AuthActionResult<{ userId: string }>>`**
   - safeParse via `signupSchema`.
   - On valid: `supabase.auth.signUp({ email, password })`. With email-confirm disabled (CONTEXT.md), Supabase returns a session immediately.
   - On Supabase error: write `auth_events { type: 'signin_failed' }` (no — for signup use a `signup_failed` event; or extend AuthEventType union with `signup_failed`. 2. **`signInAction(input: SigninInput): Promise<AuthActionResult<{ userId: string }>>`**
   - safeParse via `signinSchema`.
   - On valid: `supabase.auth.signInWithPassword(parsed.data)`.
   - On error: `writeAuthEvent({ type: 'signin_failed', email: parsed.data.email, metadata: { code: error.code } })` + return mapped error envelope. Generic-error rule via `mapSupabaseError`.
   - On success: `writeAuthEvent({ type: 'signin', userId: data.user.id, email: data.user.email })`. Add comment line `// TODO(10-05): mergeGuestCartIntoUserCart(data.user.id, sessionId)` for Plan 10-05 to extend. `revalidatePath('/', 'layout')`. Return success envelope.

3. **`signOutAction(): Promise<void>`** (no input, no AuthActionResult — UI-SPEC §"Header Changes" calls this via `<form action={signOutAction}>` which expects either redirect or void)
   - `const supabase = await createClient(); const { data: { user } } = await supabase.auth.getUser();`
   - `await supabase.auth.signOut();`
   - `await writeAuthEvent({ type: 'signout', userId: user?.id ?? null, email: user?.email ?? null });`
   - `redirect('/login')` from `next/navigation`.

4. **`requestPasswordResetAction(input: RequestResetInput): Promise<AuthActionResult<null>>`**
   - safeParse via `requestResetSchema`.
   - `await supabase.auth.resetPasswordForEmail(parsed.data.email, { redirectTo: `${origin}/api/auth/callback?next=/en/reset-password` })`.
     - Origin comes from request headers OR a hardcoded HOST (`https://orki.sa`) — use `process.env.NEXT_PUBLIC_VERCEL_URL` or fallback. Use the env helper.
   - Write `auth_events { type: 'password_reset_requested', email: parsed.data.email }`.
   - Per UI-SPEC §Anti-patterns #11: ALWAYS return `{ ok: true, data: null }` regardless of whether email exists (Supabase `resetPasswordForEmail` already does not differentiate, but if it ever errors, swallow and still return success — the UI shows the generic "if an account exists…" message).

5. **`setPasswordAction(input: SetPasswordInput): Promise<AuthActionResult<null>>`**
   - safeParse via `setPasswordSchema` (includes `refine` for password match).
   - `supabase.auth.updateUser({ password: parsed.data.password })` — the user's recovery session must be active in the cookie (set by `/api/auth/callback/route.ts` which Plan 10-04 ships).
   - On success: write `auth_events { type: 'password_changed', userId: data.user.id, email: data.user.email }`. Return `{ ok: true, data: null }`.

| Server Action → public.auth_events (Drizzle) | postgres-role write; bypasses RLS by design |
