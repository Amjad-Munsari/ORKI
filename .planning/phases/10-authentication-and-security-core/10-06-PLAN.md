<approach>
**1. `src/app/[locale]/admin/audit/page.tsx`** (per 10-PATTERNS §"`src/app/[locale]/account/page.tsx`" analog + admin-page convention):
```tsx
import { db } from '@/lib/db/client';
import { authEvents } from '@/lib/db/schema';
import { desc } from 'drizzle-orm';
import { AuditTable } from '@/components/admin/AuditTable';

export const dynamic = 'force-dynamic';

interface Props {
  params: Promise<{ locale: string }>;
  searchParams: Promise<{ page?: string }>;
}

const PAGE_SIZE = 50;

export default async function AdminAuditPage({ searchParams }: Props) {
  const sp = await searchParams;
  const page = Math.max(1, parseInt(sp.page ?? '1', 10) || 1);
  const offset = (page - 1) * PAGE_SIZE;

  const rows = await db
    .select()
    .from(authEvents)
    .orderBy(desc(authEvents.createdAt))
    .limit(PAGE_SIZE)
    .offset(offset);

  return (
    <section className="space-y-6">
      <header className="space-y-2">
        <p className="text-[12px] uppercase tracking-widest text-white/40 font-bold">Audit</p>
        <h1 className="text-4xl md:text-6xl font-bold uppercase tracking-tighter text-white leading-none">Auth events</h1>
        <p className="text-sm text-white/60">Page {page} · {rows.length} rows</p>
      </header>
      <AuditTable rows={rows} />
      <nav className="flex items-center gap-4 pt-6">
        {page > 1 && <a href={`?page=${page - 1}`} className="text-sm underline underline-offset-4">Newer</a>}
        {rows.length === PAGE_SIZE && <a href={`?page=${page + 1}`} className="text-sm underline underline-offset-4">Older</a>}
      </nav>
    </section>
  );
}
```
Drizzle is the chosen path here (postgres role connection — already bypasses RLS). The PATTERNS doc and CONTEXT.md mention using the admin client; both paths land on the same data. Drizzle is preferred because it matches the rest of the admin tree (`admin/inventory/page.tsx` uses Drizzle).
