      onPrimary?: () => void;

      primaryHref?: string;
      /** Secondary always navigates home. */
      secondaryHref?: string;
      /** Optional digest line (dev-only) shown below body. */
      digest?: string;
    }

    export function BrandedErrorPage({
      variant,
      heading,
      body,
      primaryLabel,
      secondaryLabel,
      onPrimary,
      primaryHref,
      secondaryHref = '/',
      digest,
    }: Props) {
      const role = variant === 'error' ? 'alert' : undefined;
      const PrimaryEl = onPrimary
        ? (
            <button
              onClick={onPrimary}
              className="px-6 py-3 bg-white text-black font-semibold rounded-none"

            </button>

              href={primaryHref ?? '/'}
