    **Step 1 — Create `src/components/error/BrandedErrorPage.tsx`** (Server-Component-friendly chrome consumed from both Client and RSC contexts):

    ```tsx
    import { Link } from '@/i18n/navigation';
    import type { Locale } from '@/types/domain';

    export type BrandedErrorVariant = 'error' | '404';

    export interface BrandedErrorPageProps {
      variant: BrandedErrorVariant;
      heading: string;
      body: string;
      ctaLabel: string;
      ctaHref: string;
      locale: Locale;
      isRtl: boolean;
      errorDigest?: string;
      // Variant extras:
      onPrimary?: () => void;     // when set, primary renders as <button> (error variant reset())
      secondaryLabel?: string;
      secondaryHref?: string;
    }

    export function BrandedErrorPage({
      variant,
      heading,
      body,
      ctaLabel,
      ctaHref,
      locale: _locale,           // reserved for future locale-specific tweaks; currently unused at runtime
      isRtl: _isRtl,             // reserved; outer container relies on parent layout's dir attr
      errorDigest,
      onPrimary,
      secondaryLabel,
      secondaryHref,
    }: BrandedErrorPageProps) {
      const role = variant === 'error' ? 'alert' : undefined;
      const Primary = onPrimary
        ? (
            <button
              onClick={onPrimary}
              className="px-6 py-3 bg-white text-black font-semibold rounded-none"
            >
              {ctaLabel}
            </button>
          )
        : (
            <Link
              href={ctaHref}
              className="inline-block px-6 py-3 bg-white text-black font-semibold rounded-none"
            >
              {ctaLabel}
            </Link>
          );
      return (
        <div role={role} className="min-h-[80vh] bg-black flex items-center justify-center px-6">
          <div className="max-w-md text-center space-y-8">
            <h1 className="text-6xl md:text-[100px] font-bold leading-none tracking-tighter text-white">
              {heading}
            </h1>
            <p className="text-lg text-white/60">{body}</p>
            {errorDigest && process.env.NODE_ENV === 'development' && (
              <p className="text-xs text-white/40">Error ID: {errorDigest}</p>
            )}
            <div className="flex gap-4 justify-center pt-4">
              {Primary}
              {secondaryLabel && secondaryHref && (
                <Link
                  href={secondaryHref}
                  className="inline-block px-6 py-3 border border-white/30 text-white font-semibold rounded-none"
                >
                  {secondaryLabel}
                </Link>
              )}
            </div>
          </div>
        </div>
      );
    }
    ```

    Notes: Server-Component-friendly; works inside `'use client'` files because React permits it (no `'use client'` on this chrome itself). The `<button onClick>` only fires when `onPrimary` is passed — which only happens from `error.tsx` (Client Component). `not-found.tsx` (RSC) and `global-error.tsx` (Client, no reset path used in chrome) pass `ctaHref` instead.

    import { useEffect } from 'react';
    import { geist, ibmPlexArabic } from '@/lib/fonts';
    import { BrandedErrorPage } from '@/components/error/BrandedErrorPage';

    const COPY = {
      en: {
        heading: 'Something broke.',

        retry: 'Try again',
        home: 'Return home',
      },
      ar: {

    export default function GlobalError({
      error,
      reset,
    }: {
      error: Error & { digest?: string };
      reset: () => void;

        // eslint-disable-next-line no-console

              body={t.body}

              secondaryHref="/"
