**Step 3: Create src/store/cartStore.ts** — Zustand store with persist + skipHydration. Use the curried `create<State>()()` form required by Zustand v5 for TypeScript:

```typescript
import { create } from 'zustand'
import { persist, createJSONStorage } from 'zustand/middleware'
import type { CartItem, Product } from '@/types/domain'

interface CartState {
  items: CartItem[]
  addItem: (product: Product, selectedSize: string) => void
  removeItem: (productId: string, selectedSize: string) => void
  clearCart: () => void
  getTotalCount: () => number
}

// Curried form: create<State>()() — required for TypeScript inference in Zustand v5
export const useCartStore = create<CartState>()(
  persist(
    (set, get) => ({
      items: [],

      addItem: (product, selectedSize) =>
        set(state => {
          const existing = state.items.find(
            i => i.product.id === product.id && i.selectedSize === selectedSize
          )
          if (existing) {
            return {
              items: state.items.map(i =>
                i.product.id === product.id && i.selectedSize === selectedSize
                  ? { ...i, quantity: i.quantity + 1 }
                  : i
              ),
            }
          }
          return { items: [...state.items, { product, selectedSize, quantity: 1 }] }
        }),

      removeItem: (productId, selectedSize) =>
        set(state => ({
          items: state.items.filter(
            i => !(i.product.id === productId && i.selectedSize === selectedSize)
          ),
        })),

      clearCart: () => set({ items: [] }),

      getTotalCount: () =>
        get().items.reduce((sum, i) => sum + i.quantity, 0),
    }),
    {
      name: 'orki-cart',
      storage: createJSONStorage(() => localStorage),
      skipHydration: true, // CRITICAL: prevents SSR hydration mismatch on Next.js App Router
    }
  )
)
```

**Step 4: Create src/store/StoreHydration.tsx** — Client component that triggers rehydration on mount:

```typescript
'use client'
import { useEffect } from 'react'
import { useCartStore } from './cartStore'

// Renders nothing — exists solely to trigger cart rehydration after client mount.
// useEffect only runs on the client, so localStorage access is safe.
// Must be placed inside NextIntlClientProvider in layout.tsx.
export function StoreHydration() {
  useEffect(() => {
    useCartStore.persist.rehydrate()
  }, [])

  return null
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| localStorage → CartStore | Cart data read from localStorage on client mount; no server-side access |
| CartStore → AddToCartButton/CartBadge | Client components read from store; no server components can access this |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-05 | Tampering | localStorage (orki-cart) | accept | Frontend-only phase; cart data has no monetary value until backend integration. Backend will re-validate inventory at purchase time. Document as known limitation. |
| T-02-06 | Spoofing | CartStore addItem | accept | No authentication in Phase 2; any client can call addItem. Backend validation gate is Phase 3+ concern. |
| T-02-07 | Information Disclosure | localStorage | accept | Cart contents (product names, sizes, quantities) stored in plaintext localStorage. No PII stored. Acceptable for ecommerce cart. |
</threat_model>
